type DeepPick<T, K extends string> = K extends `${infer Key}.${infer Rest}`
  ? Key extends keyof T
    ? { [P in Key]: DeepPick<T[P], Rest> }
    : never
  : K extends keyof T
    ? { [P in K]: T[K] }
    : never

  profile: {
    name: string
    email: string
  }
  settings: {
    theme: string
  }
}

type Test14 = DeepPick<User, 'profile.name'>
