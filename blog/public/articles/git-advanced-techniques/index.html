MAX_FILE_SIZE=$(git config hooks.maxfilesize)

    MAX_FILE_SIZE=5242880  # 5MB

large_files=$(git diff --cached --name-only | xargs -I{} find {} -type f -size +$MAX_FILE_SIZE 2>/dev/null)

    echo "Error: Large files detected in commit:"

    exit 1
fi
