          images: myorg/myapp

            type=ref,event=branch
            type=ref,event=pr
            type=semver,pattern={{version}}
            type=semver,pattern={{major}}.{{minor}}

          context: .

#!/bin/bash
# deploy.sh - é›¶åœæœºéƒ¨ç½²è„šæœ¬

set -e

APP_NAME="myapp"
NEW_VERSION=$1
CURRENT_CONTAINER="${APP_NAME}_current"
NEW_CONTAINER="${APP_NAME}_new"

if [ -z "$NEW_VERSION" ]; then
    echo "Usage: $0 <version>"
    exit 1
fi

echo "Deploying version $NEW_VERSION"

# æ‹‰å–æ–°é•œåƒ
docker pull myorg/$APP_NAME:$NEW_VERSION

# å¯åŠ¨æ–°å®¹å™¨
docker run -d \
    --name $NEW_CONTAINER \
    --env-file ./production.env \
    -e NODE_ENV=production \
    --network app-network \
    myorg/$APP_NAME:$NEW_VERSION

# å¥åº·æ£€æŸ¥
echo "Waiting for new container to be healthy..."
for i in {1..30}; do
    if docker exec $NEW_CONTAINER curl -f http://localhost:3000/health; then
        echo "New container is healthy"
        break
    fi
    if [ $i -eq 30 ]; then
        echo "Health check failed"
        docker logs $NEW_CONTAINER
        docker stop $NEW_CONTAINER
        docker rm $NEW_CONTAINER
        exit 1
    fi
    sleep 2
done

# åˆ‡æ¢æµé‡
docker stop $CURRENT_CONTAINER
