    this.startTime = Date.now()
  }

  measureFPS() {
    this.frameCount++
    const currentTime = Date.now()
    const elapsed = currentTime - this.startTime

    if (elapsed >= 1000) {
      const fps = Math.round((this.frameCount * 1000) / elapsed)

      this.frameCount = 0
      this.startTime = currentTime
    }
  }
}
