type BatchProducer struct {
    buffer chan Message
    size   int
    timer  *time.Timer
}

func (p *BatchProducer) Start() {
    ticker := time.NewTicker(100 * time.Millisecond)
    defer ticker.Stop()

    batch := make([]Message, 0, p.size)

    for {
        select {
        case msg := <-p.buffer:
            batch = append(batch, msg)
            if len(batch) >= p.size {
                p.send(batch)
                batch = make([]Message, 0, p.size)
            }
        case <-ticker.C:
            if len(batch) > 0 {
                p.send(batch)
                batch = make([]Message, 0, p.size)
            }
        }
    }
}
